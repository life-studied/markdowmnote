## 3.7-1 逆转字符串

```C++
#include <string>
#include <iostream>
using namespace std;

void reverse(string &s)
{
    int size = s.size();
    for (size_t i = 0; i < size / 2; i++)
    {
        char temp = s[i];
        s[i] = s[size - 1 - i];
        s[size - 1 - i] = temp;
    }
}
int main()
{
    string s("123456");
    cout << s << endl;
    reverse(s);
    cout << s << endl;
}
```
