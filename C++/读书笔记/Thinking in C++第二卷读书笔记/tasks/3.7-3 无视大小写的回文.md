## 3.7-3 无视大小写的回文

```C++
#include <string>
#include <cctype>
#include <iostream>
using namespace std;

void reverse(string &s)
{
    int size = s.size();
    for (size_t i = 0; i < size / 2; i++)
    {
        char temp = tolower(s[i]);
        s[i] = tolower(s[size - 1 - i]);
        s[size - 1 - i] = temp;
    }
}

int main(int argc, char const *argv[])
{
    if (argc < 2)
    {
        cout << "argc less than 2!" << endl;
        return -1;
    }

    string s(argv[1]);
    string s2(argv[1]);
    for (auto &i : s2)
        i = tolower(i);

    reverse(s);
    cout << (s == s2) << endl;
    return 0;
}
```
